def is_valid (maze, visited, row, col, rows, cols):
return ( 0 <= row < rows and 0 <= col < cols and
maze[row][col] == 0 and not visited[row][col])

def dfs_recursive (maze, visited, row, col, rows, cols):
if row == rows - 1 and col == cols - 1 :
return True # Ditemukan end

visited[row][col] = True

# Arah : atas, bawah, kiri, kanan
directions = [( - 1 , 0 ), ( 1 , 0 ), ( 0 , - 1 ), ( 0 , 1 )]
for dr, dc in directions :
new_row, new_col = row + dr, col + dc
if is_valid (maze, visited, new_row, new_col, rows, cols):
if dfs_recursive (maze, visited, new_row, new_col, rows, cols):
return True

return False

def solve_maze_recursive (maze):
if not maze or not maze[ 0 ]:
return False
rows, cols = len (maze), len (maze[ 0 ])
visited = [[False] * cols for _ in range (rows)]
return dfs_recursive (maze, visited, 0 , 0 , rows, cols)
